    let stdTime = [Date.parse("2022-01-13 07:54:00"), Date.parse("2022-01-13 8:02:00"), Date.parse("2022-01-13 19:54:00"), Date.parse("2022-01-13 20:02:00")];

    return 1;
    }
    return 0;
}

        itemMain.Value = curTimeValue;
        isMult = true;
